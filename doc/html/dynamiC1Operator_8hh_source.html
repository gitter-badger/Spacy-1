232  void assembleOperator(double t, const ::Spacy::Vector& x) const
233  {
234  if( b_ != nullptr && (x == oldX_F) && (t == oldT_F)) return;
235 
236  auto x_ = dolfin::Function( J_.function_space(0) );
237  copy(x,x_);
238  assign_x_if_present(F_,x_);
239  assign_t_if_present(F_,dolfin::Constant(t));
240  assign_t_if_present(f_,dolfin::Constant(t));
241  assign_f_if_present(F_,f_);
242  b_ = x_.vector()->factory().create_vector();
243 
244  dolfin::Assembler assembler;
245  assembler.assemble(*b_, F_);
246 
247  for(const auto& bc : bcs_)
248  bc->apply( *b_ , *x_.vector() );
249 
250  oldX_F = x;
251  oldT_F = t;
252  }

255  void assembleGradient(double t, const ::Spacy::Vector& x) const

265  dolfin::Assembler assembler;
