229  void assembleJacobian(const ::Spacy::Vector& x) const
230  {
231  if( b_ != nullptr && (oldX_J_==x) ) return;
232 
233  auto x_ = dolfin::Function( J_.function_space(0) );
234  copy(x,x_);
235  assign_x_if_present(J_,x_);
236  b_ = x_.vector()->factory().create_vector();
237 
238  dolfin::assemble(*b_,J_);
239  for( auto& bc : bcs_) bc->apply(*b_,*x_.vector());
240

245  void assembleHessian(const ::Spacy::Vector& x) const
246  {
247  if( A_ != nullptr && (oldX_H_==x) ) return;
248 
249  auto x_ = dolfin::Function( J_.function_space(0) );
250  copy(x,*x_.vector());
251  assign_x_if_present(H_,x_);
252  A_ = x_.vector()->factory().create_matrix();
253  dolfin::assemble(*A_,H_);
254 
255  for( auto& bc : bcs_) bc->apply(*A_,*x_.vector(),*x_.vector());
256 
257  oldX_H_ = x;
258  }
